The TabBarSkin enum skin is used with the TabBar component.
The TabPanel itself does not need an ui!
To style the Panel, add an PanelSkin enum skin. 
    <TabPanel height="150"
               title="TabPanel"
               tabBarHeaderPosition="1">
      <tabBar>
        <TabBar ui="{TabBarSkin.DEFAULT.getSkin()}"/>
      </tabBar>
      <items>
        <Panel title="Tab 1"
               html="Header position: top<br>Tab position: top"/>
        <Panel title="Tab 2"
               html="Hello World"/>
      </items>
    </TabPanel>
